async function gerarAudio() { const liveElem = document.getElementById('nota-live'); const texto = liveElem ? liveElem.innerText.trim() : ''; const btn = document.getElementById('btn-audio'); const audioBox = document.getElementById('audio-box'); const player = document.getElementById('audio-player'); const downloadBtn = document.getElementById('btn-download'); if (!texto) { return alert('Por favor, clique em "⚡ Gerar Notas de Venda" primeiro para criar o roteiro!'); } btn.innerText = '⏳ Gravando locução em MP3 via Fal.ai...'; btn.disabled = true; try { const res = await fetch('/api/generate-audio', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ texto }) }); const data = await res.json(); if (data.sucesso && data.audioUrl) { player.src = data.audioUrl; downloadBtn.href = data.audioUrl; btn.style.display = 'none'; audioBox.style.display = 'block'; } else { alert(data.erro || 'Erro ao gerar áudio.'); btn.innerText = '🎙️ Tentar Novamente'; } } catch (err) { alert('Erro de conexão ao gerar áudio.'); btn.innerText = '🎙️ Tentar Novamente'; } finally { btn.disabled = false; } }
